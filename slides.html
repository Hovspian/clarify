Clarify

  • A todo app using GTD methodologies and the Eisenhower Priority Matrix to help you focus on high-importance goals and break them down into completable tasks
  • Implemented in Elm
Forms: 
form [onSubmit CreateLifeGoal] [
    input [
        onInput UpdateCreateLifeGoalRegister
    ] [],
    button [type_ "submit"] [text "Create"]
]
type Msg =
    -- simple states
    TodayState
    ...
    -- action states
    | CreateLifeGoal
    ...

update msg model =
    case msg of
        -- every time user enters a character,
        -- we update the model
        UpdateCreateLifeGoalRegister in_text -> {
            model |
            new_life_goal_title = in_text
        }
        -- when user submits the form, we turn the
        -- current in-progress input and turn
        -- it into an actual life goal
        CreateLifeGoal -> {
            model |
            life_goals = (
                List.append
                    model.life_goals
                    [
                        LifeGoal
                            model.new_life_goal_title
                            []
                            model.lgid
                    ]
            ),
            lgid = model.lgid + 1,
            new_life_goal_title = ""
        }
        ...
